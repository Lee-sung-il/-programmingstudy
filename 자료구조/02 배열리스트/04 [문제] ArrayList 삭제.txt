/** 마지막 길이 반환 함수 ***/
int Endlen(int* arr)
{
	int len = 0, i = 0;
	while (arr[i] != '\0')	
		i++;
	return i;
}
/**  데이터 출력 함수 ***/
void Output(int* arr)
{
	int len = Endlen(arr);
	for (int i = 0; i < len; i++)
		printf("%d ", arr[i]);
	printf("\n");

}

void DSearch(int* arr, int target)
{
	//순차탐색 
	//몇회 중복되었는지
	//중복된 idx는 어디어디인지를 화면에 출력
	// [출력예]
	// 11 11 22 22 33
	// 조회값 입력 : 22
	// 22는 1,2 idx에 총 2회 중복되었습니다

	
	int cnt = 0; // 몇회중복인지 저장
	for (int i = 0; i < Endlen(arr); i++)
	{
		if (arr[i] == target)
		{
			cnt++;
			printf("%d,", i);
		}
	}
	if (cnt != 0)
		printf("\b Idx에 총 %d번 중복되었습니다\n", cnt);
	else
		printf("해당 값은 조회되지 않습니다\n");

}
